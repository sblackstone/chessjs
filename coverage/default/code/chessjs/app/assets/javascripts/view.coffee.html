(function () { (function() {
  Chess.View = (function() {
    function View() {}
 
    View.prototype.set_state = function(state) {
      this.state = state;
      return this.draw();
    };
 
    View.prototype.draw = function() {
      var i, len, ref, results, s, sq;
      console.log("view: draw");
      ref = $('.square');
      results = [];
      for (i = 0, len = ref.length; i < len; i++) {
        sq = ref[i];
        s = this.state.str_at($(sq).data("num"));
        $(sq).find('.piece').removeClass().addClass("piece").addClass(s);
        if (s === "EE") {
          s = "";
        }
        results.push($(sq).find('.piece').html(s));
      }
      return results;
    };
 
    return View;
 
  })();
 
}).call(this);
 
}());
